# AI Prowler - Python Dependencies
# Install with: pip install -r requirements.txt

# ── Core RAG / indexing ────────────────────────────────────────────────────────
chromadb==0.6.3
# sentence-transformers and huggingface-hub are pinned together deliberately.
# Mismatched versions cause huggingface_hub to return a cache path with a
# trailing backslash; sentence-transformers then appends another backslash
# via os.path.join, producing a double-backslash path that Windows rejects
# with Errno 22 Invalid argument. These two versions are tested together on
# both Windows 10 and Windows 11 and do not have this bug.
huggingface-hub==0.26.5
sentence-transformers==3.3.1
# transformers is pinned to a tested version compatible with the huggingface-hub
# and sentence-transformers pair above. Without this pin, pip backtracks through
# 20+ transformers versions at install time trying to satisfy tokenizers
# constraints from pre-existing user site-packages, making installs slow and
# nondeterministic.
transformers==4.44.2

# ── Document parsing ───────────────────────────────────────────────────────────
pdfplumber>=0.10.3
python-docx>=1.1.0
pypdf>=3.17.4
# openpyxl: reads modern .xlsx files (XML-based, Excel 2007+).
# xlrd: reads legacy .xls files (BIFF8 binary format, Excel 97-2003).
# Both are required for proper Excel indexing in rag_preprocessor.py.
# Without these, Excel files fall back to raw-binary ingestion which
# produces unreadable garbage and cannot be searched semantically.
openpyxl>=3.1.0
xlrd>=2.0.1
# python-pptx: reads modern .pptx PowerPoint files (ZIP-of-XML format).
# Without this, .pptx files cannot be indexed (binary ZIP garbage).
python-pptx>=0.6.21
# beautifulsoup4: strips HTML tags from .html/.htm/.xhtml files.
# Without this, HTML files are indexed with all tag/JS/CSS noise included.
beautifulsoup4>=4.12.0
# striprtf: removes RTF control codes from .rtf files.
# Without this, .rtf files are indexed with \rtf1\ansi\deff0 noise.
striprtf>=0.0.26
# odfpy: extracts text from OpenDocument .odt files (also a ZIP of XML).
# Without this, .odt files produce binary garbage identical to the xls problem.
odfpy>=1.4.1

# ── OCR — scanned PDFs and image files ────────────────────────────────────────
# pytesseract: Python wrapper for the Tesseract OCR engine.
#   Requires the Tesseract binary — installed automatically by AI-Prowler-Setup.iss
#   (downloads from https://github.com/UB-Mannheim/tesseract).
# pypdfium2: renders PDF pages to PIL images (no poppler needed — pure pip).
# pillow: image I/O used by both pytesseract and pypdfium2.
pytesseract>=0.3.10
pypdfium2>=4.0.0
pillow>=10.0.0

# ── Email parsing ──────────────────────────────────────────────────────────────
extract-msg>=0.45.0

# ── HTTP / API ──────────────────────────────────────────────────────────────────────────────
requests>=2.31.0

# ── HTTP MCP server (Remote Access / Cloudflare Tunnel) ──────────────────────
# mcp: official Anthropic MCP Python SDK — provides FastMCP, the tool decorator,
# and both stdio and streamable-http transports used by ai_prowler_mcp.py.
# CRITICAL: without this package the MCP server will not start.
mcp>=1.0.0
# uvicorn: ASGI server used by ai_prowler_mcp.py --transport http.
# Required to serve the streamable-http MCP endpoint that Cloudflare Tunnel exposes.
uvicorn>=0.29.0

# ── Numeric / array support ────────────────────────────────────────────────────
# numpy: required by chromadb and sentence-transformers. Listed explicitly to
# ensure a compatible version is installed before those packages pull it in.
numpy>=1.24.0

# ── GPU acceleration (auto-installed by INSTALL.bat) ──────────────────────────
# torch is intentionally NOT listed here. The installer detects whether an
# NVIDIA GPU is present and installs the correct build automatically:
#   - NVIDIA GPU found : pip install torch --index-url .../whl/cu121  (~2.5 GB)
#   - No GPU found     : pip install torch                            (~200 MB)
# To install manually with CUDA:  pip install torch --index-url https://download.pytorch.org/whl/cu121
# To install manually (CPU only): pip install torch
